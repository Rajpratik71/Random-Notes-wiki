 {$R *.res}
 
 function GetCurrentComputerName : String;<br>
 var<br>
   NameBuf: array[0..80] of Char;<br>
   SizeBuf: LongWord;<br>
 begin<br>
   SizeBuf := Sizeof(NameBuf);<br>
   GetComputerName(NameBuf, SizeBuf);<br>
   result := NameBuf;<br>
 end;<br>

   GetCurrentComputerName;<br>

 end.<br>
